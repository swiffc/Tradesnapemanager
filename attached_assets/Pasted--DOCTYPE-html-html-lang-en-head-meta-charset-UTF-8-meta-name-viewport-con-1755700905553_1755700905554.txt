                updateTimeframeInfo();

            });
        });

        function resetAnimation() {
            const bars = document.querySelectorAll('.cycle-bar');
            bars.forEach((bar, index) => {
                bar.style.animation = 'none';
                setTimeout(() => {
                    bar.style.animation = `slideIn 0.5s forwards`;
                    bar.style.animationDelay = `${index * 0.2}s`;
                }, 10);
            });
        }

        function animateCycle() {
            if (animationInProgress) return;
            animationInProgress = true;
            
            const bars = document.querySelectorAll('.cycle-bar');
            
            // Reset all
            bars.forEach(bar => {
                bar.style.opacity = '0.3';
                bar.style.transform = 'scale(1)';
            });
            
            // Animate sequence
            let step = 0;
            const animateStep = () => {
                if (step < bars.length) {
                    if (step > 0) {
                        bars[step - 1].style.opacity = '0.7';
                        bars[step - 1].style.transform = 'scale(1)';
                    }
                    
                    bars[step].style.opacity = '1';
                    bars[step].style.transform = 'scale(1.1)';
                    
                    step++;
                    setTimeout(animateStep, 1000);
                } else {
                    // Reset after animation
                    setTimeout(() => {
                        bars.forEach(bar => {
                            bar.style.opacity = '1';
                            bar.style.transform = 'scale(1)';
                        });
                        animationInProgress = false;
                    }, 500);
                }
            };
            animateStep();
        }

        // Initial animation on load
        setTimeout(() => {
            resetAnimation();
        }, 500);

        // Add event listener for animation button
        document.getElementById('animateBtn').addEventListener('click', animateCycle);
